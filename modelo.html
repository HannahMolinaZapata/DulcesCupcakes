CREATE DATABASE dulcescupcakesbd;
USE dulcescupcakesbd;

CREATE TABLE rolempresa (
    IDRol VARCHAR(10) PRIMARY KEY,
    RolEmpresa VARCHAR(20)
);

CREATE TABLE usuarios (
    IDUsuario VARCHAR(10) PRIMARY KEY,
    NombreUsu VARCHAR(255) NOT NULL,
    Correo VARCHAR(255) NOT NULL,
    Telefono VARCHAR(12) NOT NULL,
    Contrasena VARCHAR(255) NOT NULL,
    IDRol VARCHAR(10),
    FOREIGN KEY (IDRol) REFERENCES rolempresa(IDRol)
);

CREATE TABLE categorias (
    IDCategoria VARCHAR(10) PRIMARY KEY,
    Categoria VARCHAR(255) NOT NULL
);

CREATE TABLE productos (
    IDProducto VARCHAR(10) PRIMARY KEY,
    NombrePro VARCHAR(150) NOT NULL,
    Stock INT,
    Precio INT,
    Descripcion VARCHAR(400),
    IDCategoria VARCHAR(10),
    FOREIGN KEY (IDCategoria) REFERENCES categorias(IDCategoria)
);

CREATE TABLE pedido (
    IDPedido VARCHAR(10) PRIMARY KEY,
    IDUsuario VARCHAR(10),
    IDProducto VARCHAR(10),
    Cantidad INT,
    FOREIGN KEY (IDUsuario) REFERENCES usuarios(IDUsuario),
    FOREIGN KEY (IDProducto) REFERENCES productos(IDProducto)
);

CREATE TABLE detallepedido (
    IDDetallePedido VARCHAR(10) PRIMARY KEY,
    IDPedido VARCHAR(10),
    Total INT,
    FechaFactura DATE,
    CantidadTotal INT,
    Direccion VARCHAR(255),
    FOREIGN KEY (IDPedido) REFERENCES pedido(IDPedido)
);

CREATE TABLE contactanos (
    IDMensaje VARCHAR(10) PRIMARY KEY,
    IDUsuario VARCHAR(10),
    Correo VARCHAR(255) NOT NULL,
    Mensaje VARCHAR(500),
    FOREIGN KEY (IDUsuario) REFERENCES usuarios(IDUsuario)
);
